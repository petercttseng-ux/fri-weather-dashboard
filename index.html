Supabase 免費資料庫建表 SQL 
create table if not exists weather_observations (
  id bigserial primary key,
  station_id text not null,
  station_name text,
  county text,
  town text,
  observed_at timestamptz not null,
  source text not null,
  rain_1h numeric,
  rain_24h numeric,
  rain_48h numeric,
  temperature numeric,
  humidity numeric,
  wind_speed numeric,
  pressure numeric,
  payload jsonb,
  created_at timestamptz default now()
);

create index if not exists weather_observations_time_idx
  on weather_observations (observed_at);

create index if not exists weather_observations_area_idx
  on weather_observations (county, town);
